<script>
    const ADMIN_PASSWORD = 'signal123';
    
    function checkAuth() {
        if(localStorage.getItem('adminAuthenticated') !== 'true') {
            showLogin();
        } else {
            loadDashboard();
        }
    }
    
    function showLogin() {
        document.getElementById('app').innerHTML = `
            <div class="login-container">
                <h2>🔐 Admin Login</h2>
                <p>Signal Strength Inc</p>
                <input type="password" id="password" placeholder="Enter admin password">
                <button onclick="login()">Login</button>
            </div>
        `;
    }
    
    function login() {
        const pwd = document.getElementById('password').value;
        if(pwd === ADMIN_PASSWORD) {
            localStorage.setItem('adminAuthenticated', 'true');
            loadDashboard();
        } else {
            alert('Wrong password!');
        }
    }
    
    async function loadDashboard() {
        const API_URL = 'https://signal-strength-api.onrender.com';
        
        try {
            const response = await fetch(`${API_URL}/admin/transactions`);
            const transactions = await response.json();
            
            const completed = transactions.filter(t => t.status === 'completed').length;
            const pending = transactions.filter(t => t.status === 'pending').length;
            const totalAmount = transactions.reduce((sum, t) => sum + (t.amount || 0), 0);
            
            document.getElementById('app').innerHTML = `
                <div class="container">
                    <h1>📊 SIGNAL STRENGTH INC - DASHBOARD</h1>
                    <button class="refresh-btn" onclick="loadDashboard()">🔄 Refresh</button>
                    <button class="refresh-btn" onclick="logout()" style="background:#dc3545; margin-left:10px;">🚪 Logout</button>
                    
                    <div class="stats">
                        <div class="stat-card">
                            <h3>${transactions.length}</h3>
                            <p>Total Orders</p>
                        </div>
                        <div class="stat-card">
                            <h3>${completed}</h3>
                            <p>Completed</p>
                        </div>
                        <div class="stat-card">
                            <h3>${pending}</h3>
                            <p>Pending</p>
                        </div>
                        <div class="stat-card">
                            <h3>₦${totalAmount.toLocaleString()}</h3>
                            <p>Total Revenue</p>
                        </div>
                    </div>
                    
                    <table>
                        <thead>
                            <tr><th>Date</th><th>Transaction ID</th><th>Service</th><th>Package</th><th>Smartcard</th><th>Amount</th><th>Customer</th><th>Status</th></tr>
                        </thead>
                        <tbody>
                            ${transactions.map(t => `
                                <tr>
                                    <td>${new Date(t.date).toLocaleString()}</td>
                                    <td>${t.reference || t.id}</td>
                                    <td>${t.service?.toUpperCase()}</td>
                                    <td>${t.packageName}</td>
                                    <td>${t.smartcard}</td>
                                    <td>₦${(t.amount || 0).toLocaleString()}</td>
                                    <td>${t.phone}<br><small>${t.email}</small></td>
                                    <td><span class="status-${t.status}">${t.status}</span></td>
                                </tr>
                            `).join('')}
                        </tbody>
                    </table>
                </div>
            `;
        } catch(error) {
            document.getElementById('app').innerHTML = `<div class="container"><p style="color:red;">Error loading dashboard: ${error.message}</p></div>`;
        }
    }
    
    function logout() {
        localStorage.removeItem('adminAuthenticated');
        showLogin();
    }
    
    checkAuth();
    window.loadDashboard = loadDashboard;
    window.login = login;
    window.logout = logout;
</script>
